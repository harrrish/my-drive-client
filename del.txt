try {
  const { data, status } = await axiosWithCreds.post(`/directory/${folderId || ""}`, {
    folderName,
  });
  console.log("Success status code:", status);
  return data.message;
} catch (error) {
  if (axios.isAxiosError(error)) {
    // Status code of the failed request
    const statusCode = error.response?.status;
    const errorMsg = error.response?.data?.error || "Unknown error";

    console.log("Error status code:", statusCode);
    console.log("Error message:", errorMsg);

    // You can throw an object with both message and status if needed
    throw { message: errorMsg, status: statusCode };
  }
  throw { message: "Something went wrong", status: null };
}
